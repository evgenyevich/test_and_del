                        CustomerId int not null

select c.CustomerId,o.PurchaiseDatetime,o.ProductName 
       from @Customers c join 
	   @Orders o on o.CustomerId=c.CustomerId and 
	   o.CustomerId in (select CustomerId from @Orders where ProductName='молоко') and  
	   o.CustomerId not in (select CustomerId from @Orders where ProductName='сметана') and 
	   o.ProductName='молоко' 
       group by c.CustomerId,o.PurchaiseDatetime,o.ProductName
